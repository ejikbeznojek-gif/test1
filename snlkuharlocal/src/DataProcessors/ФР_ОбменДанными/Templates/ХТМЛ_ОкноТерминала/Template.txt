<!DOCTYPE html>
<html>
	<head>    
	    <meta http-equiv="Content-Type" content="text/html; charset=utf-8" />    
	    <script type="text/javascript">        

	        function write_mes(data) {
	        	p = document.createElement("div");
	        	p.innerHTML=data;
	        	
	        	el = document.getElementById('terminal_window').appendChild(p);       		        	
	        	el.scrollIntoView({block: "center", behavior: "smooth"});
	        }
	        
	        function clear(data) {
	        	p = document.createElement("div");
	        	p.innerHTML=data;
	        	el=document.getElementById('terminal_window').innerHTML="";
	        	
	        }
	    </script>
